 </xsd:annotation>
-<xsd:sequence>
-<xsd:choice>

 <xsd:documentation>Defines the set of points that represents a nominal surface. The order of the points in the aggregate is not significant relative to the surface. The order represents the "point index" which is used for assigning properties.</xsd:documentation>
 </xsd:annotation>
 </xsd:element>
